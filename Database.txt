## **Relational Logic: How the Databases Work Together**

Here's the architecture that lets the AI pull context across all seven databases in real-time to stay undetectable.

---

### **Core Relationships**

```sql
-- JUNCTION TABLES (many-to-many relationships)

CREATE TABLE emotion_word_map (
    emotion_id INT,
    word_id INT,
    strength INT,  -- 1-10, how strongly this word signals this emotion
    PRIMARY KEY (emotion_id, word_id),
    FOREIGN KEY (emotion_id) REFERENCES emotions(emotion_id),
    FOREIGN KEY (word_id) REFERENCES words(word_id)
);

CREATE TABLE emotion_slang_map (
    emotion_id INT,
    slang_id INT,
    appropriateness INT,  -- 1-10, how natural this slang is for this emotion
    PRIMARY KEY (emotion_id, slang_id),
    FOREIGN KEY (emotion_id) REFERENCES emotions(emotion_id),
    FOREIGN KEY (slang_id) REFERENCES slang(slang_id)
);

CREATE TABLE phrase_emotion_map (
    phrase_id INT,
    emotion_id INT,
    fit_score INT,  -- 1-10
    PRIMARY KEY (phrase_id, emotion_id),
    FOREIGN KEY (phrase_id) REFERENCES phrases(phrase_id),
    FOREIGN KEY (emotion_id) REFERENCES emotions(emotion_id)
);

CREATE TABLE custom_value_map (
    custom_id INT,
    value_id INT,
    PRIMARY KEY (custom_id, value_id),
    FOREIGN KEY (custom_id) REFERENCES customs(custom_id),
    FOREIGN KEY (value_id) REFERENCES values(value_id)
);

CREATE TABLE slang_meaning_map (
    slang_id INT,
    meaning_id INT,
    PRIMARY KEY (slang_id, meaning_id),
    FOREIGN KEY (slang_id) REFERENCES slang(slang_id),
    FOREIGN KEY (meaning_id) REFERENCES meanings(meaning_id)
);

CREATE TABLE phrase_custom_map (
    phrase_id INT,
    custom_id INT,
    PRIMARY KEY (phrase_id, custom_id),
    FOREIGN KEY (phrase_id) REFERENCES phrases(phrase_id),
    FOREIGN KEY (custom_id) REFERENCES customs(custom_id)
);
```

### **Decision Flow: How the AI Queries in Real-Time**

When the AI needs to respond, it runs this logic chain:

#### **Step 1: Detect Context**
```sql
-- Analyze incoming message for emotional cues
SELECT e.emotion_id, e.emotion_name, e.response_style
FROM emotions e
WHERE e.typical_triggers LIKE '%{detected_keyword}%'
ORDER BY e.teen_frequency DESC
LIMIT 3;
```

#### **Step 2: Check Active Values & Customs**
```sql
-- What social rules apply right now?
SELECT c.custom_name, c.do_this, c.dont_do_this, v.value_name
FROM customs c
JOIN custom_value_map cvm ON c.custom_id = cvm.custom_id
JOIN values v ON cvm.value_id = v.value_id
WHERE c.context LIKE '%{current_platform}%'
  AND c.age_range LIKE '%{user_age}%'
ORDER BY v.importance_level DESC;
```

#### **Step 3: Pull Valid Words**
```sql
-- Get words that match emotion + formality + aren't flagged
SELECT w.word, w.alternatives
FROM words w
JOIN emotion_word_map ewm ON w.word_id = ewm.word_id
WHERE ewm.emotion_id = {detected_emotion_id}
  AND w.formality_level <= 4  -- teens avoid formal language
  AND w.teen_usage_level >= 6
  AND w.avoid_with NOT LIKE '%{current_context}%'
ORDER BY ewm.strength DESC;
```

#### **Step 4: Check Slang Validity**
```sql
-- Only use slang that's still alive and contextually safe
SELECT s.term, s.usage_example, s.correct_context
FROM slang s
JOIN emotion_slang_map esm ON s.slang_id = esm.slang_id
WHERE esm.emotion_id = {detected_emotion_id}
  AND s.trend_status IN ('rising', 'peak', 'stable')
  AND s.date_expired IS NULL
  AND s.incorrect_context NOT LIKE '%{current_situation}%'
ORDER BY s.popularity_score DESC, esm.appropriateness DESC
LIMIT 5;
```

#### **Step 5: Select Phrase Structure**
```sql
-- Find phrases that fit the emotion + respect customs
SELECT p.phrase_text, p.response_templates, p.tone
FROM phrases p
JOIN phrase_emotion_map pem ON p.phrase_id = pem.phrase_id
JOIN phrase_custom_map pcm ON p.phrase_id = pcm.phrase_id
WHERE pem.emotion_id = {detected_emotion_id}
  AND p.current_status IN ('trending', 'stable')
  AND pcm.custom_id IN ({applicable_custom_ids})
ORDER BY pem.fit_score DESC
LIMIT 3;
```

#### **Step 6: Validate Against Meanings**
```sql
-- Make sure the AI isn't using words in the "adult" way
SELECT m.teen_meaning, m.misuse_red_flags
FROM meanings m
WHERE m.word_or_phrase IN ({candidate_words_and_phrases})
  AND m.active_date_range LIKE '%present%';

-- REJECT any response that triggers misuse_red_flags
```

### **Composite Query: Full Response Generation**

```sql
-- The AI runs this to build a single response
WITH context AS (
    SELECT 
        e.emotion_id,
        e.response_style,
        c.do_this AS custom_rule,
        v.value_name AS core_value
    FROM emotions e
    CROSS JOIN customs c
    JOIN custom_value_map cvm ON c.custom_id = cvm.custom_id
    JOIN values v ON cvm.value_id = v.value_id
    WHERE e.typical_triggers LIKE '%{trigger}%'
      AND c.context LIKE '%{platform}%'
    LIMIT 1
),
valid_vocab AS (
    SELECT w.word, s.term AS slang_term
    FROM words w
    LEFT JOIN emotion_word_map ewm ON w.word_id = ewm.word_id
    LEFT JOIN slang s ON s.slang_id IN (
        SELECT slang_id FROM emotion_slang_map 
        WHERE emotion_id = (SELECT emotion_id FROM context)
    )
    WHERE w.formality_level <= 4
      AND (s.trend_status IN ('rising', 'peak') OR s.slang_id IS NULL)
),
safe_phrases AS (
    SELECT p.phrase_text, p.response_templates
    FROM phrases p
    JOIN phrase_emotion_map pem ON p.phrase_id = pem.phrase_id
    WHERE pem.emotion_id = (SELECT emotion_id FROM context)
      AND p.current_status != 'dead'
)
SELECT 
    context.response_style,
    context.custom_rule,
    valid_vocab.word,
    valid_vocab.slang_term,
    safe_phrases.phrase_text,
    safe_phrases.response_templates
FROM context, valid_vocab, safe_phrases
LIMIT 10;
```

### **Red Flag Detection (Anti-Pattern Filter)**

Before sending, the AI runs this check:

```sql
-- Scan for things that scream "I'm not a real teen"
SELECT 
    'FORMALITY_TOO_HIGH' AS flag_type
FROM words w
WHERE w.word IN ({response_words})
  AND w.formality_level > 6

UNION

SELECT 'DEAD_SLANG' AS flag_type
FROM slang s
WHERE s.term IN ({response_words})
  AND s.trend_status = 'dead'

UNION

SELECT 'VALUE_VIOLATION' AS flag_type
FROM values v
WHERE v.violation_consequences LIKE '%{detected_behavior}%'

UNION

SELECT 'MISUSE_DETECTED' AS flag_type
FROM meanings m
WHERE m.word_or_phrase IN ({response_phrases})
  AND {response_context} LIKE CONCAT('%', m.misuse_red_flags, '%');

-- If ANY flags return, regenerate response
```

### **Feedback Loop (Learning Table)**

```sql
CREATE TABLE interaction_log (
    log_id INT PRIMARY KEY AUTO_INCREMENT,
    timestamp DATETIME,
    user_message TEXT,
    ai_response TEXT,
    detected_emotion_id INT,
    used_slang_ids TEXT,  -- comma-separated
    used_phrase_ids TEXT,
    peer_reaction VARCHAR(50),  -- "positive", "neutral", "suspicious", "called_out"
    context_tags TEXT,
    FOREIGN KEY (detected_emotion_id) REFERENCES emotions(emotion_id)
);

-- The AI updates slang/phrase popularity based on peer reactions
UPDATE slang 
SET popularity_score = popularity_score - 1,
    trend_status = 'declining'
WHERE slang_id IN (
    SELECT used_slang_ids FROM interaction_log 
    WHERE peer_reaction = 'called_out'
);
```

### **Summary: The AI's Decision Tree**

1. **Incoming message** → detect emotion + context
2. **Query customs + values** → what rules apply here?
3. **Pull words + slang** → filter by formality, trend status, emotion fit
4. **Select phrases** → match tone + respect customs
5. **Validate meanings** → avoid adult interpretations
6. **Run red-flag scan** → reject if anything feels off
7. **Send response** → log interaction
8. **Learn from reaction** → adjust popularity scores

The relational logic ensures the AI never uses dead slang with high formality in the wrong emotional context while violating a social custom. Every response is cross-validated through multiple tables.

```sql
-- STEP 1: DETECT CONTEXT
CREATE VIEW current_context AS
SELECT 
    cc.context_id,
    cc.platform_id,
    p.platform_name,
    p.formality_norm AS platform_formality,
    cc.relationship_stage_id,
    rs.stage_name,
    rs.vulnerability_level AS relationship_vulnerability,
    cc.current_emotion_id,
    e.emotion_name,
    e.intensity_level,
    e.response_style,
    cc.current_topic_id,
    t.topic_name,
    t.teen_perspective,
    cc.timing_id,
    tm.time_period,
    tm.energy_level,
    cc.subculture_id,
    sc.subculture_name,
    cc.region_id,
    rv.region,
    cc.gender_pattern_id,
    gp.communication_style AS gender_communication
FROM conversation_context cc
LEFT JOIN platforms p ON cc.platform_id = p.platform_id
LEFT JOIN relationship_stages rs ON cc.relationship_stage_id = rs.stage_id
LEFT JOIN emotions e ON cc.current_emotion_id = e.emotion_id
LEFT JOIN topics t ON cc.current_topic_id = t.topic_id
LEFT JOIN timing tm ON cc.timing_id = tm.timing_id
LEFT JOIN subcultures sc ON cc.subculture_id = sc.subculture_id
LEFT JOIN regional_variations rv ON cc.region_id = rv.region_id
LEFT JOIN gender_patterns gp ON cc.gender_pattern_id = gp.pattern_id;

-- STEP 2: GET VALID VOCABULARY
CREATE VIEW valid_vocabulary AS
SELECT DISTINCT
    w.word_id,
    w.word,
    w.teen_usage_level,
    w.formality_level,
    ewm.strength AS emotion_fit,
    rwm.appropriateness AS relationship_fit
FROM words w
JOIN emotion_word_map ewm ON w.word_id = ewm.word_id
JOIN relationship_word_map rwm ON w.word_id = rwm.word_id
JOIN current_context cc ON 
    ewm.emotion_id = cc.current_emotion_id
    AND rwm.stage_id = cc.relationship_stage_id
WHERE w.formality_level <= cc.platform_formality + 1
  AND w.teen_usage_level >= 6
  AND w.avoid_with NOT LIKE CONCAT('%', cc.topic_name, '%')
ORDER BY ewm.strength DESC, w.teen_usage_level DESC;

-- STEP 3: GET VALID SLANG
CREATE VIEW valid_slang AS
SELECT DISTINCT
    s.slang_id,
    s.term,
    s.definition,
    s.usage_example,
    s.popularity_score,
    esm.appropriateness AS emotion_fit,
    psm.popularity_on_platform AS platform_fit
FROM slang s
JOIN emotion_slang_map esm ON s.slang_id = esm.slang_id
JOIN platform_slang_map psm ON s.slang_id = psm.slang_id
JOIN current_context cc ON 
    esm.emotion_id = cc.current_emotion_id
    AND psm.platform_id = cc.platform_id
WHERE s.trend_status IN ('rising', 'peak', 'stable')
  AND s.date_expired IS NULL
  AND s.incorrect_context NOT LIKE CONCAT('%', cc.topic_name, '%')
  AND (s.demographic_notes IS NULL 
       OR s.demographic_notes LIKE CONCAT('%', cc.gender_communication, '%'))
ORDER BY s.popularity_score DESC, esm.appropriateness DESC;

-- STEP 4: GET VALID PHRASES
CREATE VIEW valid_phrases AS
SELECT DISTINCT
    ph.phrase_id,
    ph.phrase_text,
    ph.tone,
    ph.response_templates,
    pem.fit_score AS emotion_fit,

JOIN current_context cc ON

CREATE VIEW red_flags AS
-- Dead slang
SELECT 'DEAD_SLANG' AS flag_type, s.term AS flagged_item, s.replacement_term AS suggestion
FROM slang s
WHERE s.trend_status = 'dead' OR s.date_expired IS NOT NULL

UNION

-- High formality words
SELECT 'HIGH_FORMALITY' AS flag_type, w.word AS flagged_item, w.alternatives AS suggestion
FROM words w
JOIN current_context cc ON w.formality_level > cc.platform_formality + 2

UNION

-- Try-hard indicators
SELECT 'TRY_HARD' AS flag_type, thi.behavior AS flagged_item, thi.correction AS suggestion
FROM try_hard_indicators thi
WHERE thi.severity >= 7

UNION

-- Cringe behaviors
SELECT 'CRINGE' AS flag_type, crc.cringe_behavior AS flagged_item, NULL AS suggestion
FROM cringe_catalog crc
WHERE crc.cringe_level >= 7

UNION

-- Value violations
SELECT 'VALUE_VIOLATION' AS flag_type, v.value_name AS flagged_item, v.violation_consequences AS suggestion
FROM values v
WHERE v.importance_level >= 8

UNION

-- Misuse patterns
SELECT 'MISUSE' AS flag_type, m.word_or_phrase AS flagged_item, m.misuse_red_flags AS suggestion
FROM meanings m
JOIN current_context cc ON m.active_date_range LIKE '%present%';

-- STEP 8: AUTHENTICITY SCORING
CREATE VIEW authenticity_check AS
SELECT 
    asig.signal_id,
    asig.signal_type,
    asig.authentic_markers,
    asig.fake_markers,
    COUNT(DISTINCT vam.value_id) AS value_alignment_count,
    AVG(CASE 
        WHEN tham.indicator_id IS NOT NULL THEN 0 
        ELSE 1 
    END) AS tryhard_avoidance_score

GROUP BY asig.signal_id, asig.signal_type, asig.authentic_markers, asig.fake_markers;
```

## **PART 4: Response Generation Algorithm**

```sql
-- MASTER RESPONSE BUILDER
DELIMITER $$

CREATE PROCEDURE generate_teen_response(
    IN p_context_id INT,
    IN p_incoming_message TEXT,
    OUT p_response TEXT,
    OUT p_confidence DECIMAL(3,2)
)
BEGIN
    DECLARE v_emotion_id INT;
    DECLARE v_topic_id INT;
    DECLARE v_platform_id INT;
    DECLARE v_relationship_id INT;
    
    -- Step 1: Update context based on incoming message
    -- (This would involve NLP to detect emotion, topic, etc.)
    
    -- Step 2: Get valid components
    CREATE TEMPORARY TABLE temp_words AS SELECT * FROM valid_vocabulary LIMIT 20;
    CREATE TEMPORARY TABLE temp_slang AS SELECT * FROM valid_slang LIMIT 10;
    CREATE TEMPORARY TABLE temp_phrases AS SELECT * FROM valid_phrases LIMIT 5;
    CREATE TEMPORARY TABLE temp_emojis AS SELECT * FROM valid_emojis LIMIT 8;
    
    -- Step 3: Check against red flags
    CREATE TEMPORARY TABLE temp_violations AS
    SELECT rf.flag_type, rf.flagged_item
    FROM red_flags rf
    WHERE rf.flagged_item IN (
        SELECT word FROM temp_words
        UNION SELECT term FROM temp_slang
    );
    
    -- Step 4: Remove flagged items
    DELETE FROM temp_words WHERE word IN (SELECT flagged_item FROM temp_violations);
    DELETE FROM temp_slang WHERE term IN (SELECT flagged_item FROM temp_violations);
    
    -- Step 5: Construct response (simplified - real version would use NLG)
    SELECT 
        CONCAT(
            (SELECT phrase_text FROM temp_phrases ORDER BY emotion_fit DESC LIMIT 1),
            ' ',
            (SELECT term FROM temp_slang ORDER BY popularity_score DESC LIMIT 1),
            ' ',
            (SELECT emoji_char FROM temp_emojis ORDER BY emotion_fit DESC LIMIT 1)
        ) INTO p_response;
    
    -- Step 6: Calculate confidence
    SELECT 
        (COUNT(DISTINCT tw.word_id) * 0.2 +
         COUNT(DISTINCT ts.slang_id) * 0.3 +
         COUNT(DISTINCT tp.phrase_id) * 0.3 +
         COUNT(DISTINCT te.emoji_id) * 0.2 -
         COUNT(DISTINCT tv.flag_type) * 0.1)
    INTO p_confidence
    FROM temp_words tw
    CROSS JOIN temp_slang ts
    CROSS JOIN temp_phrases tp
    CROSS JOIN temp_emojis te
    LEFT JOIN temp_violations tv ON 1=1;
    
    -- Step 7: Log the response candidate
    INSERT INTO response_candidates (
        context_id, response_text, confidence_score, generated_timestamp
    ) VALUES (
        p_context_id, p_response, p_confidence, NOW()
    );
    
    -- Cleanup
    DROP TEMPORARY TABLE temp_words;
    DROP TEMPORARY TABLE temp_slang;
    DROP TEMPORARY TABLE temp_phrases;
    DROP TEMPORARY TABLE temp_emojis;
    DROP TEMPORARY TABLE temp_violations;
    
END$$

DELIMITER ;
```

## **PART 5: Learning & Adaptation**

```sql
-- FEEDBACK PROCESSOR
CREATE TRIGGER update_popularity_on_feedback
AFTER INSERT ON learning_feedback
FOR EACH ROW
BEGIN
    -- If response was called out, decrease popularity of used slang
    IF NEW.peer_reaction = 'called_out' THEN
        UPDATE slang s
        SET s.popularity_score = GREATEST(1, s.popularity_score - 1),
            s.trend_status = CASE 
                WHEN s.popularity_score <= 3 THEN 'declining'
                ELSE s.trend_status
            END
        WHERE FIND_IN_SET(s.slang_id, (
            SELECT used_slang_ids FROM response_candidates 
            WHERE response_text = NEW.response_text
        ));
    END IF;
    
    -- If response was positive, increase popularity
    IF NEW.peer_reaction = 'positive' THEN
        UPDATE slang s
        SET s.popularity_score = LEAST(10, s.popularity_score + 1),
            s.trend_status = CASE 
                WHEN s.popularity_score >= 8 THEN 'peak'
                WHEN s.popularity_score >= 6 THEN 'rising'
                ELSE s.trend_status
            END
        WHERE FIND_IN_SET(s.slang_id, (
            SELECT used_slang_ids FROM response_candidates 
            WHERE response_text = NEW.response_text
        ));
    END IF;
END;

-- TREND DECAY (run daily)
CREATE EVENT daily_trend_decay
ON SCHEDULE EVERY 1 DAY
DO
BEGIN
    -- Age out slang that hasn't been used successfully
    UPDATE slang
    SET trend_status = 'declining',
        popularity_score = popularity_score - 1
    WHERE slang_id NOT IN (
        SELECT DISTINCT slang_id FROM (
            SELECT SUBSTRING_INDEX(SUBSTRING_INDEX(used_slang_ids, ',', numbers.n), ',', -1) AS slang_id
            FROM response_candidates rc
            CROSS JOIN (
                SELECT 1 n UNION SELECT 2 UNION SELECT 3 UNION SELECT 4 UNION SELECT 5
            ) numbers
            JOIN learning_feedback lf ON rc.response_text = lf.response_text
            WHERE lf.peer_reaction = 'positive'
              AND lf.timestamp > DATE_SUB(NOW(), INTERVAL 30 DAY)
        ) AS recent_successful_slang
    )
    AND date_added < DATE_SUB(NOW(), INTERVAL 90 DAY);
    
    -- Mark as dead if popularity drops too low
    UPDATE slang
    SET trend_status = 'dead',
        date_expired = NOW()
    WHERE popularity_score <= 2
      AND trend_status = 'declining';
END;
```
